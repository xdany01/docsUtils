(static) isEmpty(value) → {boolean}

Description:

  • Verifica si un valor está vacío.

    Un valor se considera vacío si:

    • Es una cadena de texto y está vacía después de eliminar espacios en blanco.
    • Es un array y no tiene elementos.
    • Es un objeto y no tiene propiedades propias.
    • Es un número (incluyendo float, bigint e integer) y es igual a 0.
Source:
  • validators-fn.js, line 172
Example
console.log(Validator.isEmpty("   "));   // true (cadena vacía)
console.log(Validator.isEmpty([]));      // true (array vacío)
console.log(Validator.isEmpty({}));      // true (objeto vacío)
console.log(Validator.isEmpty(0));       // true (número 0)
console.log(Validator.isEmpty(3.14));    // false (número distinto de 0)
console.log(Validator.isEmpty("Hello")); // false (cadena con contenido)
console.log(Validator.isEmpty([1, 2]));  // false (array con elementos)
console.log(Validator.isEmpty({ a: 1 })); // false (objeto con propiedades)
Parameters:
Name Type Description
value *

El valor a evaluar.
Returns:

true si el valor está vacío, false en caso contrario.

(static) validateObject(obj, schema) → {Object}

Description:

  • Valida un objeto según un esquema de validación.

Source:
  • validators-fn.js, line 352
Example
// Ejemplo de uso con código postal y coordenadas
const objeto = {
    codigo_postal: "12345",   // Código postal válido
    latitud: 40.7128,         // Latitud válida
    longitud: -74.0060        // Longitud válida
};
const esquema = {
    codigo_postal: {required: true, type: "string", regex: /^\d{5}$/}, // Código postal de 5 dígitos
    latitud: {required: true, type: "number"}, // Latitud debe ser un número
    longitud: {required: true, type: "number"} // Longitud debe ser un número
};
console.log(validateObject(objeto, esquema));
// { valido: true }

// Ejemplo de uso con código postal y coordenadas
// no valido por campo requerido
const objeto = {
    codigo_postal: "12345",   // Código postal válido
    latitud: 40.7128,         // Latitud válida
    // longitud: -74.0060        // Longitud omitida
};
const esquema = {
    codigo_postal: {required: true, type: "string", regex: /^\d{5}$/}, // Código postal de 5 dígitos
    latitud: {required: true, type: "number"}, // Latitud debe ser un número
    longitud: {required: true, type: "number"} // Longitud debe ser un número
};
console.log(validateObject(objeto, esquema));
// { valido: false, errores: [ 'Falta la clave obligatoria: longitud' ] }

// Ejemplo de uso con código postal y coordenadas
// no valido por regex
const objeto = {
    codigo_postal: "123",   // Código postal no cumple la regex
    latitud: 40.7128,         // Latitud válida
    longitud: -74.0060        // Longitud válida
};
const esquema = {
    codigo_postal: {required: true, type: "string", regex: /^\d{5}$/}, // Código postal de 5 dígitos
    latitud: {required: true, type: "number"}, // Latitud debe ser un número
    longitud: {required: true, type: "number"} // Longitud debe ser un número
};
console.log(validateObject(objeto, esquema));
// {
//   valido: false,
//   errores: [
//     'El valor de "codigo_postal" no cumple con la expresión regular /^\\d{5}$/'
//   ]
// }

// Ejemplo de uso con código postal y coordenadas
// no valido por longitud
const objeto = {
    codigo_postal: "123",     // Código postal no cumple la regex
    latitud: 40.7128,         // Latitud válida
    longitud: -74.0060        // Longitud válida
};
const esquema = {
    codigo_postal: {required: true, type: "string", minlength: 5}, // Código postal de 5 dígitos
    latitud: {required: true, type: "number", minlength: 50.25415}, // Latitud debe ser un número
    longitud: {required: true, type: "number"} // Longitud debe ser un número
};
console.log(validateObject(objeto, esquema));
// {
//     valido: false,
//         errores: [
//     'El valor de "codigo_postal" no cumple con la con la longitud mínima de 5',
//     'El valor de "latitud" no cumple con la con la longitud mínima de 50.25415'
// ]
// }
