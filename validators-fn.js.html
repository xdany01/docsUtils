/**
 * Clase Validator que proporciona métodos estáticos para validar tipos de datos.
 */
export class Validator {
    /**
     * Verifica si un valor es un string.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un string, de lo contrario false.
     * @example
     * Validator.isString("Hola"); // true
     */
    static isString(value) {
        return typeof value === 'string'; // Comprobación del tipo de dato string
    }

    /**
     * Verifica si un valor es un número.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un número, de lo contrario false.
     * @example
     * Validator.isNumber(42); // true
     */
    static isNumber(value) {
        return typeof value === 'number'; // Comprobación del tipo de dato number
    }

    /**
     * Verifica si un valor es un entero.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un entero, de lo contrario false.
     * @example
     * Validator.isInteger(42); // true
     */
    static isInteger(value) {
        return Number.isInteger(value); // Comprobación de si el número es un entero
    }

    /**
     * Verifica si un valor es un número de punto flotante.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un número de punto flotante, de lo contrario false.
     * @example
     * Validator.isFloat(3.14); // true
     */
    static isFloat(value) {
        return this.isNumber(value) && !this.isInteger(value); // Comprobación de número de punto flotante
    }

    /**
     * Verifica si un valor es un booleano.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un booleano, de lo contrario false.
     * @example
     * Validator.isBoolean(false); // true
     */
    static isBoolean(value) {
        return typeof value === 'boolean'; // Comprobación del tipo de dato boolean
    }

    /**
     * Verifica si un valor es una función.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es una función, de lo contrario false.
     * @example
     * Validator.isFunction(() => {}); // true
     */
    static isFunction(value) {
        return typeof value === 'function'; // Comprobación del tipo de dato function
    }

    /**
     * Verifica si un valor es un objeto (pero no un array o null).
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un objeto, de lo contrario false.
     * @example
     * Validator.isObject({}); // true
     */
    static isObject(value) {
        return value !== null && typeof value === 'object' && !Array.isArray(value); // Comprobación de objeto
    }

    /**
     * Verifica si un valor es un array.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un array, de lo contrario false.
     * @example
     * Validator.isArray([]); // true
     */
    static isArray(value) {
        return Array.isArray(value); // Comprobación de array
    }

    /**
     * Verifica si un valor es null.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es null, de lo contrario false.
     * @example
     * Validator.isNull(null) // true
     */
    static isNull(value) {
        return value === null; // Comprobación de null
    }

    /**
     * Verifica si un valor es undefined.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es undefined, de lo contrario false.
     * @example
     * Validator.isUndefined(undefined); // true
     */
    static isUndefined(value) {
        return typeof value === 'undefined'; // Comprobación de undefined
    }

    /**
     * Verifica si un valor es un símbolo.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un símbolo, de lo contrario false.
     * @example
     * Validator.isSymbol(Symbol()); // true
     */
    static isSymbol(value) {
        return typeof value === 'symbol'; // Comprobación del tipo de dato symbol
    }

    /**
     * Verifica si un valor es un BigInt.
     * @param {any} value - El valor a verificar.
     * @returns {boolean} - Retorna true si el valor es un BigInt, de lo contrario false.
     * @example
     * Validator.isBigInt(42n); // true
     */
    static isBigInt(value) {
        return typeof value === 'bigint'; // Comprobación del tipo de dato bigint
    }

    /**
     * Verifica que el indice se encuentre dentro de los valores permitidos para la matriz suministrada.
     * @param {Number} index
     * @param {any[]} arr
     * @returns {boolean} Retorna `true` si esta dentro del rango y `false` si no lo esta.
     */
    static indexInRange(index, arr) {
        return index >= 0 && index < arr.length - 1
    }

    /**
     * Verifica si la variable es del tipo indicado.
     * @param {*} value Variable a testear.
     * @param {string} type Tipo de dato en formato string.
     * @returns {boolean} Retorna `true` si es del tipo indicado y `false` si no es.
     */
    static isType(value, type) {
        return typeof value !== type
    }

    /**
     * Comprueba si un objeto está vacío.
     * @param {{}} obj Objeto a comprobar.
     * @returns {boolean} Retorna `true` si esta vacio y `false` si no lo esta.
     */
    static isEmptyObj(obj) {
        return Reflect.ownKeys(obj).length === 0 && obj.constructor === Object;
    }

    /**
     * Verifica si el tipo indicado se encuentra de los tipos validos javascript.
     * @param {string} type Tipo de dato en formato string.
     * @returns {boolean} Retorna `true` si es un tipo valido y `false` si no es.
     */
    static isValidType(type) {
        const primitive = ['number', 'string', 'boolean', 'null', 'undefined'];
        const complex = ['date', 'datetime', 'datetime-local', 'object', 'array', 'function', 'regexp', 'Map', 'Set'];
        const specials = ['symbol', 'bigint'];
        const types = [...primitive, ...complex, ...specials];
        return types.includes(type);
    }

    /**
     * Permite determinar si la matriz es unidimensional (es decir, contiene elementos directamente y no otras matrices).
     * @param {any[]} arr Matriz que se desea verificar.
     * @returns {boolean} Retorna `true` si es unidimensional y `false` si no lo es.
     * @example
     * Validator.isOneDimensionalMatrix([1, 2, 3, 4, 5]); // true
     * Validator.isOneDimensionalMatrix([[1, 2, 3], [4, 5, 6]]); // false
     */
    static isOneDimensional(arr) {
        let isArr = false;
        for (let elem of arr) {
            isArr = Array.isArray(elem);
            if (isArr) {
                break;
            }
        }
        return !isArr;
    }

    /**
     * Permite validar que un objeto tenga una propiedad determinada y que sea de tipo indicado.
     * Si no se indica el tipo verificara que sea del tipo `RegExp`.
     * @param {Object} obj Objeto a verificar propiedad.
     * @param {string} prop Propiedad a verificar que exista en el objeto.
     * @param {string|null} type=null Tipo de dato.
     * @returns {boolean} Retorna `true` si es valido y `false` si no lo es.
     * @example
     * const miObjeto = {
     *   regex1: /\d+/,
     *   regex2: /^[a-zA-Z]+$/,
     *   otraPropiedad: 'Hola Mundo'
     * };
     * Validator.checkDataTypeObj(miObjeto, 'regex1'); // true
     * Validator.checkDataTypeObj(miObjeto, 'regex2'); // true
     * Validator.checkDataTypeObj(miObjeto, 'otraPropiedad'); // false
     * Validator.checkDataTypeObj(miObjeto, 'propiedadInexistente'); // false
     * Validator.checkDataTypeObj(miObjeto, 'regex1', 'object'); // true (es una expresión regular)
     * Validator.checkDataTypeObj(miObjeto, 'regex2', 'object'); // true (es una expresión regular)
     * Validator.checkDataTypeObj(miObjeto, 'otraPropiedad', 'string'); // true
     * Validator.checkDataTypeObj(miObjeto, 'propiedadInexistente', 'string'); // false (la propiedad no existe)
     * Validator.checkDataTypeObj(miObjeto, 'otraPropiedad', 'object'); // false (la propiedad no es del tipo object)
     */
    static checkDataTypeObj(obj, prop, type = null) {
        let valid = false;
        if (obj.hasOwnProperty(prop)) {
            if (type === null) {
                valid = (obj[prop] instanceof RegExp);
            } else {
                valid = (typeof obj[prop] === type);
            }
        }
        return valid;
    }

    /**
     * Permite comparar si dos matrices unidimensionales son iguales.
     * @param {any[]} arr1
     * @param {any[]} arr2
     * @returns {boolean} Retorna `true` si son iguales y `false` si no lo son.
     * @example
     * Validator.equalArrays([1,2,3,4],[1,2,3,4]) // true
     */
    static equalArrays(arr1, arr2) {
        if (arr1.length !== arr2.length) return false;
        return arr1.every((valor, indice) => valor === arr2[indice]);
    }

    /**
     * Verifica si una cadena está codificada en formato URI.
     * @param {string} str - La cadena a verificar.
     * @returns {boolean} - Retorna true si la cadena está codificada, false en caso contrario.
     * @throws {Error} - Lanza un error si el argumento no es un string.
     */
    static isEncoded(str) {
        // Verifica si el tipo de argumento es un string
        if (typeof str !== 'string') {
            throw new Error('El argumento no es un string.'); // Lanza un error si no es un string
        }
        // Verifica si la cadena está vacía o solo contiene espacios
        if (!str.trim()) {
            return false; // Considera una cadena vacía o solo espacios como no codificada
        }
        try {
            // Intenta decodificar la cadena y compara con la original
            return decodeURIComponent(str) !== str; // Retorna true si la cadena es diferente después de decodificar
        } catch (e) {
            return false; // Si no se puede decodificar, asumimos que no está correctamente codificado
        }
    }
}
