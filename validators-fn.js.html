/**
 * @module Validators
 */

/**
 * Valida si un numero es un entero.
 * @param {Number} num Numero a validar.
 * @returns {boolean} Retorna `true` si es un entero y `false` si no lo es.
 */
export function isInteger(num) {
    return num % 1 === 0 && Math.floor(num) === num && Math.ceil(num) === num && Number.isInteger(num);
}

/**
 * Función para verificar si el valor ingresado es un número.
 * @param {any} n - El valor a verificar.
 * @returns {boolean} - Devuelve true si es un número, de lo contrario false.
 * @example
 * console.log(isNumber(null)); // false
 * console.log(isNumber(undefined)); // false
 * console.log(isNumber('a')); // false
 * console.log(isNumber('10')); // false
 * console.log(isNumber(10)); // true
 * console.log(isNumber(10.5)); // true
 */
export function isNumber(n) {
    // Convertir el valor a un número de punto flotante
    const num = parseFloat(n);
    // Verificar si el número no es NaN, es finito y es igual al valor original
    return !Number.isNaN(num) && Number.isFinite(num) && Number(n) === num;
}


/**
 * Verifica si un valor es undefined o null.
 * @param {*} value Variable a analizar.
 * @returns {boolean} Retorna `true` si es undefined o null y `false` si no lo es.
 */
export function isUndefinedOrNull(value) {
    return value === undefined || value === null;
}

/**
 * Verifica si el argumento es una matriz.
 * @param {*} arr Variable a testear.
 * @returns {boolean} Retorna `true` si es una matriz y `false` si no lo es.
 */
export function isArray(arr) {
    return Array.isArray(arr);
}

/**
 * Verifica que el indice se encuentre dentro de los valores permitidos para la matriz suministrada.
 * @param {Number} index
 * @param {any[]} arr
 * @returns {boolean} Retorna `true` si esta dentro del rango y `false` si no lo esta.
 */
export function indexInRange(index, arr) {
    return index >= 0 && index < arr.length - 1
}

/**
 * Verifica si la variable es del tipo indicado.
 * @param {*} value Variable a testear.
 * @param {string} type Tipo de dato en formato string.
 * @returns {boolean} Retorna `true` si es del tipo indicado y `false` si no es.
 */
export function isType(value, type) {
    return typeof value !== type
}

/**
 * Comprueba si un objeto está vacío.
 * @param {{}} obj Objeto a comprobar.
 * @returns {boolean} Retorna `true` si esta vacio y `false` si no lo esta.
 */
export function isEmptyObj(obj) {
    return Reflect.ownKeys(obj).length === 0 && obj.constructor === Object;
}

/**
 * Verifica si el tipo indicado se encuentra de los tipos validos javascript.
 * @param {string} type Tipo de dato en formato string.
 * @returns {boolean} Retorna `true` si es un tipo valido y `false` si no es.
 */
export function isValidType(type) {
    const primitive = ['number', 'string', 'boolean', 'null', 'undefined'];
    const complex = ['date', 'datetime', 'datetime-local', 'object', 'array', 'function', 'regexp', 'Map', 'Set'];
    const specials = ['symbol', 'bigint'];
    const types = [...primitive, ...complex, ...specials];
    return types.includes(type);
}

/**
 * Permite determinar si la matriz es unidimensional (es decir, contiene elementos directamente y no otras matrices).
 * @param {any[]} arr Matriz que se desea verificar.
 * @returns {boolean} Retorna `true` si es unidimensional y `false` si no lo es.
 * @example
 * isOneDimensionalMatrix([1, 2, 3, 4, 5]);
 * // true
 * isOneDimensionalMatrix([[1, 2, 3], [4, 5, 6]]);
 * // false
 */
export function isOneDimensional(arr) {
    let isArr = false;
    for (let elem of arr) {
        isArr = Array.isArray(elem);
        if (isArr) {
            break;
        }
    }
    return !isArr;
}

/**
 * Permite comparar si dos matrices unidimensionales son iguales.
 * @param {any[]} arr1
 * @param {any[]} arr2
 * @returns {boolean} Retorna `true` si son iguales y `false` si no lo son.
 * @example
 * equalArrays([1,2,3,4],[1,2,3,4])
 * // true
 */
export function equalArrays(arr1, arr2) {
    if (arr1.length !== arr2.length) return false;
    return arr1.every((valor, indice) => valor === arr2[indice]);
}

/**
 * Permite validar que un objeto tenga una propiedad determinada y que sea de tipo indicado.
 * Si no se indica el tipo verificara que sea del tipo `RegExp`.
 * @param {Object} obj Objeto a verificar propiedad.
 * @param {string} prop Propiedad a verificar que exista en el objeto.
 * @param {string} type=null Tipo de dato.
 * @returns {boolean} Retorna `true` si es valido y `false` si no lo es.
 * @example
 * const miObjeto = {
 *   regex1: /\d+/,
 *   regex2: /^[a-zA-Z]+$/,
 *   otraPropiedad: 'Hola Mundo'
 * };
 *
 * console.log(verificarExpresionRegular(miObjeto, 'regex1')); // true
 * console.log(verificarExpresionRegular(miObjeto, 'regex2')); // true
 * console.log(verificarExpresionRegular(miObjeto, 'otraPropiedad')); // false
 * console.log(verificarExpresionRegular(miObjeto, 'propiedadInexistente')); // false
 *
 * console.log(verificarTipoDeDato(miObjeto, 'regex1', 'object')); // true (es una expresión regular)
 * console.log(verificarTipoDeDato(miObjeto, 'regex2', 'object')); // true (es una expresión regular)
 * console.log(verificarTipoDeDato(miObjeto, 'otraPropiedad', 'string')); // true
 * console.log(verificarTipoDeDato(miObjeto, 'propiedadInexistente', 'string')); // false (la propiedad no existe)
 * console.log(verificarTipoDeDato(miObjeto, 'otraPropiedad', 'object')); // false (la propiedad no es del tipo object)
 */
export function checkDataTypeObj(obj, prop, type = null) {
    let valid = false;
    if (obj.hasOwnProperty(prop)) {
        if (type === null) {
            valid = (obj[prop] instanceof RegExp);
        } else {
            valid = (typeof obj[prop] === type);
        }
    }
    return valid;
}
