/**
 * Clase que representa una tarea asincrónica encadenable similar a una Promise.
 * @example
 * // Ejemplo de uso 1: Operaciones síncronas encadenadas
 * async function ejemploSincrono() {
 *     const resultado = await Task.from(5)
 *         .map(x => x * 2)
 *         .map(x => x + 3)
 *         .map(x => `El resultado es: ${x}`)
 *         .run();
 *
 *     console.log(resultado); // "El resultado es: 13"
 * }
 *
 * // Ejemplo de uso 2: Operaciones asíncronas encadenadas
 * async function ejemploAsincrono() {
 *     const simularDB = (id) =>
 *         new Promise(resolve =>
 *             setTimeout(() => resolve({ id, nombre: 'Usuario ' + id }), 1000));
 *
 *     const resultado = await Task.from(1)
 *         .map(id => simularDB(id))
 *         .map(usuario => {
 *             return Task.from(new Promise(resolve =>
 *                 setTimeout(() => resolve({
 *                     ...usuario,
 *                     edad: 25
 *                 }), 500)
 *             ));
 *         })
 *         .tap(usuario => console.log('Procesando usuario:', usuario))
 *         .map(usuario => `${usuario.nombre} tiene ${usuario.edad} años`)
 *         .catch(error => `Error: ${error.message}`)
 *         .run();
 *
 *     console.log(resultado);
 * }
 *
 * // Ejemplo de uso 3: Manejo de errores
 * async function ejemploErrores() {
 *     const resultadoConError = await Task.from(10)
 *         .map(x => {
 *             if (x > 5) throw new Error('Número muy grande');
 *             return x * 2;
 *         })
 *         .catch(error => `Capturado: ${error.message}`)
 *         .run();
 *
 *     console.log(resultadoConError); // "Capturado: Número muy grande"
 * }
 *
 * // Ejecutamos los ejemplos
 * async function ejecutarEjemplos() {
 *     console.log('=== Ejemplo Síncrono ===');
 *     await ejemploSincrono();
 *
 *     console.log('\n=== Ejemplo Asíncrono ===');
 *     await ejemploAsincrono();
 *
 *     console.log('\n=== Ejemplo con Errores ===');
 *     await ejemploErrores();
 * }
 *
 * ejecutarEjemplos();
 */
class Task {
    /**
     * Crea una nueva instancia de Task.
     * @param {function} executor - Función que ejecutará la tarea. Recibe dos parámetros: `resolve` y `reject`.
     */
    constructor(executor) {
        if (typeof executor !== 'function') {
            throw new TypeError('El executor debe ser una función');
        }
        /**
         * Función que ejecutará la tarea.
         * @type {function}
         */
        this.executor = executor;
    }

    /**
     * Crea una nueva instancia de Task a partir de un valor o promesa.
     * @param {any} value - Valor o promesa a convertir en Task.
     * @returns {Task} Nueva instancia de Task.
     */
    static from(value) {
        return new Task((resolve, reject) => {
            Promise.resolve(value).then(resolve).catch(reject);
        });
    }

    /**
     * Ejecuta la tarea y devuelve una promesa.
     * @returns {Promise<any>} Promesa con el resultado de la tarea.
     */
    run() {
        return new Promise(this.executor);
    }

    /**
     * Aplica una función al resultado de la tarea y devuelve una nueva Task.
     * @param {function(any): (any|Promise<any>|Task)} fn - Función de transformación.
     * @returns {Task} Nueva instancia de Task con la función aplicada.
     */
    map(fn) {
        return new Task((resolve, reject) => {
            this.run()
                .then(result => {
                    try {
                        const nextTask = fn(result);
                        return Promise.resolve(
                            nextTask instanceof Task ? nextTask.run() : nextTask
                        );
                    } catch (error) {
                        return Promise.reject(error);
                    }
                })
                .then(resolve)
                .catch(reject);
        });
    }

    /**
     * Maneja errores en la ejecución de la tarea.
     * @param {function(Error): any} errorHandler - Función que maneja el error y devuelve un valor de recuperación.
     * @returns {Task} Nueva instancia de Task con el manejo de errores aplicado.
     */
    catch(errorHandler) {
        return new Task((resolve, reject) => {
            this.run()
                .then(resolve)
                .catch(error => {
                    try {
                        resolve(errorHandler(error));
                    } catch (err) {
                        reject(err);
                    }
                });
        });
    }

    /**
     * Ejecuta una acción sin modificar el resultado de la tarea.
     * @param {function(any): void} action - Función a ejecutar con el resultado de la tarea.
     * @returns {Task} Nueva instancia de Task con la acción aplicada.
     */
    tap(action) {
        return this.map(result => {
            try {
                action(result);
            } catch (error) {
                console.error('Error en tap:', error);
            }
            return result;
        });
    }
}
