/**
 * @module Objects
 */

/**
 * Ordena una serie de objetos por clave dada.
 * @param {Object[]} arr Matriz de objetos que se desea ordenar.
 * @param {string} key Clave por la cual se desea ordenar la matriz.
 * @returns {Object[]} Retorna una matriz con los objetos ordenados por la clave suministrada.
 * @example
 * const people = [
 *     { name: 'Foo', age: 42 },
 *     { name: 'Bar', age: 24 },
 *     { name: 'Fuzz', age: 36 },
 *     { name: 'Baz', age: 32 },
 * ];
 * sortBy(people, 'age');
 *
 * //  [
 * //      { name: 'Bar', age: 24 },
 * //      { name: 'Baz', age: 32 },
 * //      { name: 'Fuzz', age: 36 },
 * //      { name: 'Foo', age: 42 },
 * //  ]
 */
export function sortBy(arr, key) {
    return arr.concat().sort((a, b) => (a[key] > b[key] ? 1 : a[key] < b[key] ? -1 : 0));
}

/**
 * Permite encontrar el elemento mínimo de una matriz de objetos según la clave dada.
 * @param {Object[]} arr Matriz de objetos en la cual se desea realizar la busqueda.
 * @param {string} key Clave por la cual se desea buscar.
 * @returns {Object} Retorna objeto que es el elemento mínimo de la matriz
 * de acuerdo a la clave suministrada.
 * @example
 * const people = [
 *     { name: 'Bar', age: 24 },
 *     { name: 'Baz', age: 32 },
 *     { name: 'Foo', age: 42 },
 *     { name: 'Fuzz', age: 36 },
 * ];
 * minBy(people, 'age');
 * // { name: 'Bar', age: 24 }
 */
export function minBy(arr, key) {
    return arr.reduce((a, b) => (a[key] < b[key] ? a : b), {});
}

/**
 * Permite encontrar el elemento máximo de una matriz de objetos según la clave dada.
 * @param {Object[]} arr Matriz de objetos en la cual se desea realizar la busqueda.
 * @param {string} key Clave por la cual se desea buscar.
 * @returns {Object} Retorna objeto que es el elemento máximo de la matriz
 * de acuerdo a la clave suministrada.
 * @example
 * const people = [
 *     { name: 'Bar', age: 24 },
 *     { name: 'Baz', age: 32 },
 *     { name: 'Foo', age: 42 },
 *     { name: 'Fuzz', age: 36 },
 * ];
 * maxBy(people, 'age');
 * // { name: 'Foo', age: 42 }
 */
export function maxBy(arr, key) {
    return arr.reduce((a, b) => (a[key] >= b[key] ? a : b), {});
}

/**
 * Permite contar por las propiedades de una serie de objetos.
 * @param {Object[]} arr Matriz de objetos en la cual se desea realizar la busqueda.
 * @param prop Propiedad por la cual se desea buscar.
 * @returns {Object} Retorna objeto con el valor buscado y la cantidad de coincidencias.
 * @example
 * countBy(
 *     [
 *         { branch: 'audi', model: 'q8', year: '2019' },
 *         { branch: 'audi', model: 'rs7', year: '2020' },
 *         { branch: 'ford', model: 'mustang', year: '2019' },
 *         { branch: 'ford', model: 'explorer', year: '2020' },
 *         { branch: 'bmw', model: 'x7', year: '2020' },
 *     ],
 *     'branch'
 * );
 *
 * // { 'audi': 2, 'ford': 2, 'bmw': 1 }
 */
export function countBy(arr, prop) {
    return arr.reduce((prev, curr) => ((prev[curr[prop]] = ++prev[curr[prop]] || 1), prev), {});
}

/**
 * Comprueba si dos objetos son iguales.
 * @param {Object[]} objects Suceción de objetos a comparar.
 * @returns {boolean} Retorna `true` si los objetos son iguales y `false` si no los son.
 * @example
 * isEqual({ foo: 'bar' }, { foo: 'bar' });
 * // true
 * isEqual({ foo: 'bar' }, { bar: 'foo' });
 * // false
 */
export function isEqualObj(...objects) {
    return objects.every((obj) => JSON.stringify(obj) === JSON.stringify(objects[0]));
}

/**
 * Permite extraer valores de una propiedad de una matriz de objetos.
 * @param {Object[]} objs Matriz de objetos base.
 * @param {string} property Propiedad que se busca en la matriz de objetos.
 * @returns {any[]} Retorna una matriz con los valores buscados.
 * @example
 * pluck(
 *     [
 *         { name: 'John', age: 20 },
 *         { name: 'Smith', age: 25 },
 *         { name: 'Peter', age: 30 },
 *     ],
 *     'name'
 * );
 * // ['John', 'Smith', 'Peter']
 */
export function pluck(objs, property) {
    return objs.map((obj) => obj[property]);
}

/**
 * Permite cambiar el nombre de las claves de objetos de forma inmutable.
 * @param {{}} keysMap Objeto base.
 * @param {{}} obj Objeto con las nuevas claves.
 * @returns {{}} Retorna un objeto con las claves cambiadas.
 * @example
 * const obj = { a: 1, b: 2, c: 3 };
 * const keysMap = { a: 'd', b: 'e', c: 'f' };
 * renameKeys(keysMap, obj);
 * // { d: 1, e: 2, f: 3 }
 */
export function renameKeys(keysMap, obj) {
    return Object.keys(obj).reduce((acc, key) => ({...acc, ...{[keysMap[key] || key]: obj[key]}}), {});
}

/**
 * Elimina todas las propiedades nulas e indefinidas de un objeto.
 * @param {Object} obj Objeto con entradas nulas o indefinidas.
 * @returns {Object} Retorna el objeto sin undefined y null.
 * @example
 * removeNullUndefined({
 *     foo: null,
 *     bar: undefined,
 *     fuzz: 42,
 * });
 * // { fuzz: 42 }
 */
export function removeNullUndefined(obj) {
    return Object.entries(obj)
        .filter(([_, v]) => v != null)
        .reduce((acc, [k, v]) => ({...acc, [k]: v}), {});
}

/**
 * Ordena un objeto por sus propiedades.
 * @param {Object} obj Objeto a ordenar por sus propiedades.
 * @returns {Object} Retorna un objeto ordenado por sus propiedades.
 * @example
 * const colors = {
 *     white: '#ffffff',
 *     black: '#000000',
 *     red: '#ff0000',
 *     green: '#008000',
 *     blue: '#0000ff',
 * };
 * sort(colors);
 * //{
 * //    black: '#000000',
 * //    blue: '#0000ff',
 * //    green: '#008000',
 * //    red: '#ff0000',
 * //    white: '#ffffff',
 * //}
 */
export function sortObj(obj) {
    return Object.keys(obj)
        .sort()
        .reduce((p, c) => ((p[c] = obj[c]), p), {});
}

/**
 * Permite agrupar una serie de objetos por una clave.
 * @param {Object[]} arr Matriz de objetos en la cual se desea trabajar.
 * @param {string} key Clave por la cual se desea agrupar.
 * @returns {Object} Retorna objeto clave valor agrupado por las claves.
 * @example
 * groupBy(
 *     [
 *         { branch: 'audi', model: 'q8', year: '2019' },
 *         { branch: 'audi', model: 'rs7', year: '2020' },
 *         { branch: 'ford', model: 'mustang', year: '2019' },
 *         { branch: 'ford', model: 'explorer', year: '2020' },
 *         { branch: 'bmw', model: 'x7', year: '2020' },
 *     ],
 *     'branch'
 * );
 *
 * /*
 * {
 *     audi: [
 *         { branch: 'audi', model: 'q8', year: '2019' },
 *         { branch: 'audi', model: 'rs7', year: '2020' }
 *     ],
 *     bmw: [
 *         { branch: 'bmw', model: 'x7', year: '2020' }
 *     ],
 *     ford: [
 *         { branch: 'ford', model: 'mustang', year: '2019' },
 *         { branch: 'ford', model: 'explorer', year: '2020' }
 *     ],
 * }
 */
export function groupBy(arr, key) {
    return arr.reduce((acc, item) => ((acc[item[key]] = [...(acc[item[key]] || []), item]), acc), {});
}

/**
 * Permite convertir una matriz de objetos en un solo objeto.
 * @param {Object[]} arr Matriz de objetos.
 * @param {string} key Clave por la cual se desea agrupar los objetos.
 * @returns {*} Retorna una matriz de objetos agrupados por la clave indicada.
 * @example
 * toObject(
 *     [
 *         { id: '1', name: 'Alpha', gender: 'Male' },
 *         { id: '2', name: 'Bravo', gender: 'Male' },
 *         { id: '3', name: 'Charlie', gender: 'Female' },
 *     ],
 *     'id'
 * );
 * /*
 * {
 *     '1': { id: '1', name: 'Alpha', gender: 'Male' },
 *     '2': { id: '2', name: 'Bravo', gender: 'Male' },
 *     '3': { id: '3', name: 'Charlie', gender: 'Female' },
 * }
 */
export function toObject(arr, key) {
    return Object.fromEntries(arr.map((it) => [it[key], it]));
}
