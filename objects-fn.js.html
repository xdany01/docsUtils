/**

/**
 * Ordena una serie de objetos por clave dada.
 * @param {Object[]} arr Matriz de objetos que se desea ordenar.
 * @param {string} key Clave por la cual se desea ordenar la matriz.
 * @returns {Object[]} Retorna una matriz con los objetos ordenados por la clave suministrada.
 * @example
 * const people = [
 *     { name: 'Foo', age: 42 },
 *     { name: 'Bar', age: 24 },
 *     { name: 'Fuzz', age: 36 },
 *     { name: 'Baz', age: 32 },
 * ];
 * sortBy(people, 'age');
 *
 * //  [
 * //      { name: 'Bar', age: 24 },
 * //      { name: 'Baz', age: 32 },
 * //      { name: 'Fuzz', age: 36 },
 * //      { name: 'Foo', age: 42 },
 * //  ]
 */
export function sortBy(arr, key) {
    return arr.concat().sort((a, b) => (a[key] > b[key] ? 1 : a[key] < b[key] ? -1 : 0));
}

/**
 * Permite encontrar el elemento mínimo de una matriz de objetos según la clave dada.
 * @param {Object[]} arr Matriz de objetos en la cual se desea realizar la busqueda.
 * @param {string} key Clave por la cual se desea buscar.
 * @returns {Object} Retorna objeto que es el elemento mínimo de la matriz
 * de acuerdo a la clave suministrada.
 * @example
 * const people = [
 *     { name: 'Bar', age: 24 },
 *     { name: 'Baz', age: 32 },
 *     { name: 'Foo', age: 42 },
 *     { name: 'Fuzz', age: 36 },
 * ];
 * minBy(people, 'age');
 * // { name: 'Bar', age: 24 }
 */
export function minBy(arr, key) {
    return arr.reduce((a, b) => (a[key] < b[key] ? a : b), {});
}

/**
 * Permite encontrar el elemento máximo de una matriz de objetos según la clave dada.
 * @param {Object[]} arr Matriz de objetos en la cual se desea realizar la busqueda.
 * @param {string} key Clave por la cual se desea buscar.
 * @returns {Object} Retorna objeto que es el elemento máximo de la matriz
 * de acuerdo a la clave suministrada.
 * @example
 * const people = [
 *     { name: 'Bar', age: 24 },
 *     { name: 'Baz', age: 32 },
 *     { name: 'Foo', age: 42 },
 *     { name: 'Fuzz', age: 36 },
 * ];
 * maxBy(people, 'age');
 * // { name: 'Foo', age: 42 }
 */
export function maxBy(arr, key) {
    return arr.reduce((a, b) => (a[key] >= b[key] ? a : b), {});
}

/**
 * Permite contar por las propiedades de una serie de objetos.
 * @param {Object[]} arr Matriz de objetos en la cual se desea realizar la busqueda.
 * @param prop Propiedad por la cual se desea buscar.
 * @returns {Object} Retorna objeto con el valor buscado y la cantidad de coincidencias.
 * @example
 * countBy(
 *     [
 *         { branch: 'audi', model: 'q8', year: '2019' },
 *         { branch: 'audi', model: 'rs7', year: '2020' },
 *         { branch: 'ford', model: 'mustang', year: '2019' },
 *         { branch: 'ford', model: 'explorer', year: '2020' },
 *         { branch: 'bmw', model: 'x7', year: '2020' },
 *     ],
 *     'branch'
 * );
 *
 * // { 'audi': 2, 'ford': 2, 'bmw': 1 }
 */
export function countBy(arr, prop) {
    return arr.reduce((prev, curr) => ((prev[curr[prop]] = ++prev[curr[prop]] || 1), prev), {});
}

/**
 * Comprueba si dos objetos son iguales.
 * @param {Object[]} objects Suceción de objetos a comparar.
 * @returns {boolean} Retorna `true` si los objetos son iguales y `false` si no los son.
 * @example
 * isEqual({ foo: 'bar' }, { foo: 'bar' });
 * // true
 * isEqual({ foo: 'bar' }, { bar: 'foo' });
 * // false
 */
export function isEqualObj(...objects) {
    return objects.every((obj) => JSON.stringify(obj) === JSON.stringify(objects[0]));
}

/**
 * Permite extraer valores de una propiedad de una matriz de objetos.
 * @param {Object[]} objs Matriz de objetos base.
 * @param {string} property Propiedad que se busca en la matriz de objetos.
 * @returns {any[]} Retorna una matriz con los valores buscados.
 * @example
 * pluck(
 *     [
 *         { name: 'John', age: 20 },
 *         { name: 'Smith', age: 25 },
 *         { name: 'Peter', age: 30 },
 *     ],
 *     'name'
 * );
 * // ['John', 'Smith', 'Peter']
 */
export function pluck(objs, property) {
    return objs.map((obj) => obj[property]);
}

/**
 * Permite cambiar el nombre de las claves de objetos de forma inmutable.
 * @param {{}} keysMap Objeto base.
 * @param {{}} obj Objeto con las nuevas claves.
 * @returns {{}} Retorna un objeto con las claves cambiadas.
 * @example
 * const obj = { a: 1, b: 2, c: 3 };
 * const keysMap = { a: 'd', b: 'e', c: 'f' };
 * renameKeys(keysMap, obj);
 * // { d: 1, e: 2, f: 3 }
 */
export function renameKeys(keysMap, obj) {
    return Object.keys(obj).reduce((acc, key) => ({...acc, ...{[keysMap[key] || key]: obj[key]}}), {});
}

/**
 * Elimina todas las propiedades nulas e indefinidas de un objeto.
 * @param {Object} obj Objeto con entradas nulas o indefinidas.
 * @returns {Object} Retorna el objeto sin undefined y null.
 * @example
 * removeNullUndefined({
 *     foo: null,
 *     bar: undefined,
 *     fuzz: 42,
 * });
 * // { fuzz: 42 }
 */
export function removeNullUndefined(obj) {
    return Object.entries(obj)
        .filter(([_, v]) => v != null)
        .reduce((acc, [k, v]) => ({...acc, [k]: v}), {});
}

/**
 * Ordena un objeto por sus propiedades.
 * @param {Object} obj Objeto a ordenar por sus propiedades.
 * @returns {Object} Retorna un objeto ordenado por sus propiedades.
 * @example
 * const colors = {
 *     white: '#ffffff',
 *     black: '#000000',
 *     red: '#ff0000',
 *     green: '#008000',
 *     blue: '#0000ff',
 * };
 * sort(colors);
 * //{
 * //    black: '#000000',
 * //    blue: '#0000ff',
 * //    green: '#008000',
 * //    red: '#ff0000',
 * //    white: '#ffffff',
 * //}
 */
export function sortObj(obj) {
    return Object.keys(obj)
        .sort()
        .reduce((p, c) => ((p[c] = obj[c]), p), {});
}

/**
 * Permite agrupar una serie de objetos por una clave.
 * @param {Object[]} arr Matriz de objetos en la cual se desea trabajar.
 * @param {string} key Clave por la cual se desea agrupar.
 * @returns {Object} Retorna objeto clave valor agrupado por las claves.
 * @example
 * groupBy(
 *     [
 *         { branch: 'audi', model: 'q8', year: '2019' },
 *         { branch: 'audi', model: 'rs7', year: '2020' },
 *         { branch: 'ford', model: 'mustang', year: '2019' },
 *         { branch: 'ford', model: 'explorer', year: '2020' },
 *         { branch: 'bmw', model: 'x7', year: '2020' },
 *     ],
 *     'branch'
 * );
 *
 * /*
 * {
 *     audi: [
 *         { branch: 'audi', model: 'q8', year: '2019' },
 *         { branch: 'audi', model: 'rs7', year: '2020' }
 *     ],
 *     bmw: [
 *         { branch: 'bmw', model: 'x7', year: '2020' }
 *     ],
 *     ford: [
 *         { branch: 'ford', model: 'mustang', year: '2019' },
 *         { branch: 'ford', model: 'explorer', year: '2020' }
 *     ],
 * }
 */
export function groupBy(arr, key) {
    return arr.reduce((acc, item) => ((acc[item[key]] = [...(acc[item[key]] || []), item]), acc), {});
}

/**
 * Permite convertir una matriz de objetos en un solo objeto.
 * @param {Object[]} arr Matriz de objetos.
 * @param {string} key Clave por la cual se desea agrupar los objetos.
 * @returns {*} Retorna una matriz de objetos agrupados por la clave indicada.
 * @example
 * toObject(
 *     [
 *         { id: '1', name: 'Alpha', gender: 'Male' },
 *         { id: '2', name: 'Bravo', gender: 'Male' },
 *         { id: '3', name: 'Charlie', gender: 'Female' },
 *     ],
 *     'id'
 * );
 * /*
 * {
 *     '1': { id: '1', name: 'Alpha', gender: 'Male' },
 *     '2': { id: '2', name: 'Bravo', gender: 'Male' },
 *     '3': { id: '3', name: 'Charlie', gender: 'Female' },
 * }
 */
export function toObject(arr, key) {
    return Object.fromEntries(arr.map((it) => [it[key], it]));
}

/**
 * Función para dividir las claves en partes según los separadores dados.
 * @param {Array} keys - Claves a dividir.
 * @param {Array} separators - Separadores para dividir las claves. Por defecto: ['@', '.0.']
 * @returns {Array} - Arreglo con las claves divididas.
 */
export function splitObjInKeys(keys, separators = ['@', '.0.']) {
    const arrKeys = []; // Inicializa un arreglo para almacenar las claves divididas
    if (Array.isArray(keys)) { // Verifica si las claves son un arreglo
        keys.forEach(item => { // Itera sobre cada elemento del arreglo
            let shouldSplit = false;
            separators.forEach(separator => { // Itera sobre cada separador
                if (item.includes(separator)) { // Verifica si la clave contiene el separador actual
                    item.split(separator).forEach(e => arrKeys.push(e)); // Divide la clave en caso de contener el separador y agrega las partes al arreglo
                    shouldSplit = true;
                }
            });

            if (!shouldSplit) {
                arrKeys.push(item); // Agrega la clave al arreglo si no contiene ninguno de los separadores
            }
        });
    } else {
        arrKeys.push(keys); // Agrega la clave al arreglo si no es un arreglo
    }
    return arrKeys; // Devuelve el arreglo con las claves divididas
}

/**
 * Función que obtiene un objeto JSON de un string que contiene un objeto a partir de una regex.
 * @param {string} key - La clave que puede contener un objeto JSON.
 * @param {RegExp} reg - Expresión regular para buscar el objeto JSON. Por defecto, busca cualquier objeto JSON entre paréntesis.
 * @returns {Object|null} - Devuelve el objeto JSON encontrado o null si no se encuentra.
 */
export function getObjFromKey(key, reg = new RegExp(/\((\{.*?\})\)/, 's')) {
    let jsonMatch = key.match(reg);
    return jsonMatch ? JSON.parse(jsonMatch[1]) : null;
}

/**
 * Función para combinar objetos con arreglos.
 * @param {Array} arrayOfObjects - Arreglo de objetos a combinar
 * @returns {Object} - Objeto combinado
 */
export function combinedObjectWithArrays(arrayOfObjects) {
    // Inicializar el acumulador como un objeto vacío
    return arrayOfObjects.reduce((accumulator, currentObject) => {
        // Iterar sobre las claves del objeto actual
        Object.keys(currentObject).forEach(key => {
            // Verificar si la clave no existe en el acumulador
            if (!accumulator[key]) {
                // Si no existe, crear un nuevo arreglo vacío en el acumulador
                accumulator[key] = [];
            }
            // Agregar el valor correspondiente al arreglo en el acumulador
            accumulator[key].push(currentObject[key]);
        });
        // Retornar el acumulador actualizado
        return accumulator;
    }, {});
}

/**
 * Función para eliminar duplicados de los valores de un objeto.
 * @param {object} inputObj - Objeto de entrada con posibles duplicados.
 * @returns {object} - Objeto resultante sin duplicados.
 */
export function removeDuplicatesFromValues(inputObj) {
    const result = {};
    // Iterar sobre las claves del objeto de entrada
    Object.keys(inputObj).forEach(key => {
        // Verificar si el valor es un array
        if (Array.isArray(inputObj[key])) {
            // Verificar si el primer elemento del array es un objeto y no nulo
            if (typeof inputObj[key][0] === 'object' && inputObj[key][0] !== null) {
                // Eliminar duplicados de objetos en el array
                result[key] = inputObj[key].map(JSON.stringify)
                    .filter((value, index, self) => self.indexOf(value) === index)
                    .map(JSON.parse);
            } else {
                // Eliminar duplicados simples en el array
                result[key] = Array.from(new Set(inputObj[key]));
            }
        } else {
            // Conservar el valor si no es un array
            result[key] = inputObj[key];
        }
    });
    return result;
}

/**
 * Función para eliminar arrays de los valores de un objeto.
 * @param {Object} inputObj - Objeto de entrada con posibles arrays en sus valores.
 * @returns {Object} - Objeto resultante con arrays convertidos a valores únicos si tienen longitud 1.
 */
export function removeArrayFromValues(inputObj) {
    const result = {};
    // Iterar sobre las claves del objeto de entrada
    Object.keys(inputObj).forEach(key => {
        // Verificar si el valor es un array
        if (Array.isArray(inputObj[key])) {
            // Si el array tiene longitud 1, asignar el valor único a la clave
            if (inputObj[key].length === 1) {
                result[key] = inputObj[key][0];
            } else {
                // Si el array tiene más de un elemento, mantener el array en la clave
                result[key] = inputObj[key];
            }
        } else {
            // Si no es un array, mantener el valor original en la clave
            result[key] = inputObj[key];
        }
    });
    return result;
}

/**
 * Función que encuentra las claves comunes en un array de objetos.
 * @param {Array} arr - El array de objetos de entrada.
 * @returns {Array} - Un array con las claves comunes encontradas.
 * @example
 * let arreglo = [
 *     { id: 1, nombre: 'Juan', edad: 30 },
 *     { id: 2, nombre: 'María', pais: 'Argentina' },
 *     { id: 3, nombre: 'Pedro', edad: 25, pais: 'España' }
 * ];
 *
 * let keys = obtenerKeysComunes(arreglo);
 * // Resultado:
 * ['id', 'nombre']
 */
export function commonKeys(arr) {
    // Verifica si el array está vacío y devuelve un array vacío si es así.
    if (arr.length === 0) return [];
    // Inicializa un conjunto con las claves del primer objeto del array.
    let common = new Set(Object.keys(arr[0]));
    // Itera sobre cada objeto en el array para encontrar las claves comunes.
    arr.forEach(objeto => {
        // Filtra las claves que no están presentes en el objeto actual.
        common = new Set([...common].filter(key => Object.keys(objeto).includes(key)));
    });
    // Convierte el conjunto de claves comunes a un array y lo devuelve.
    return Array.from(common);
}

/**
 * Función para eliminar de forma segura una clave de un objeto anidado.
 * @param {Object} nestedObject - El objeto anidado del cual se eliminará la clave.
 * @param {string} keyToRemove - La clave que se eliminará del objeto anidado.
 * @returns {Object} - El objeto anidado con la clave eliminada de forma segura.
 * @example
 * const nestedObj = {
 *   obj1: { a: 1, b: 2 },
 *   obj2: { c: 3, d: 4 },
 *   obj3: { e: 5, f: 6 }
 * };
 * const updatedObj = removeObjectSafely(nestedObj, 'obj2');
 * console.log(updatedObj);
 * // Output: { obj1: { a: 1, b: 2 }, obj3: { e: 5, f: 6 } }
 */
export function removeObjectSafely(nestedObject, keyToRemove) {
    // Verifica si el objeto anidado tiene la clave a eliminar
    if (nestedObject.hasOwnProperty(keyToRemove)) {
        // Elimina la clave del objeto anidado
        delete nestedObject[keyToRemove];
    }
    // Devuelve el objeto anidado modificado
    return nestedObject;
}

/**
 * Función que devuelve las claves comunes de un array de objetos.
 * @param {Array} arr - El array de objetos del cual se quieren encontrar las claves comunes.
 * @returns {Array} - Un array con las claves comunes de los objetos.
 * @example
 * const objetos = [
 *     { nombre: "Juan", edad: 25, ciudad: "Madrid" },
 *     { nombre: "Ana", apellido: "Perez", edad: 28 },
 *     { nombre: "Luis", edad: 30, ciudad: "Barcelona" }
 * ];
 * const commonKeys = getCommonKeys(objetos);
 * console.log(commonKeys);
 * // ["nombre", "edad"]
 */
export function getCommonKeys(arr) {
    // Verifica si el array está vacío y devuelve un array vacío en ese caso
    if (arr.length === 0) return [];
    // Obtiene las claves del primer objeto y filtra aquellas que están presentes en todos los objetos del array
    return Object.keys(arr[0]).filter(key =>
        arr.every(obj => key in obj)
    );
}

/**
 * Función que elimina elementos duplicados de un array basado en una propiedad específica.
 * @param {Array} array - El array del cual se eliminarán los duplicados.
 * @param {string} property - La propiedad basada en la cual se eliminarán los duplicados.
 * @returns {Array} - Un nuevo array sin elementos duplicados.
 */
export function removeDuplicatesByProperty(array, property) {
    const hash = {}; // Objeto para almacenar las propiedades únicas
    return array.filter((current) => {
        const value = current[property]; // Valor de la propiedad actual
        const exist = !hash[value]; // Verificar si la propiedad ya existe en el hash
        hash[value] = true; // Marcar la propiedad como existente en el hash
        return exist; // Devolver true si la propiedad es única, false si es duplicada
    });
}
