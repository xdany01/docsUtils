/**

 *
 * @example
 * const rotator = new UserAgentRotator('mobile');

    /**
     * Filtra los user agents según el tipo especificado.
     * @param {string[]} userAgents - Lista de user agents.
     * @param {string} type - Tipo de filtro.
     * @returns {string[]} Lista filtrada de user agents.
     */
    filterUserAgents(userAgents, type) {
        switch (type) {
            case UserAgentType.DESKTOP:
                return userAgents.filter(ua =>
                    !ua.includes('Mobile') &&
                    !ua.includes('Android') &&
                    !ua.includes('iPhone') &&
                    !ua.includes('iPad') &&
                    !ua.includes('bot')
                );

            case UserAgentType.MOBILE:
                return userAgents.filter(ua =>
                    ua.includes('Mobile') ||
                    ua.includes('iPhone') ||
                    (ua.includes('Android') && ua.includes('Mobile'))
                );

            case UserAgentType.TABLET:
                return userAgents.filter(ua =>
                    ua.includes('iPad') ||
                    (ua.includes('Android') && !ua.includes('Mobile'))
                );

            case UserAgentType.ANDROID:
                return userAgents.filter(ua => ua.includes('Android'));

            case UserAgentType.IOS:
                return userAgents.filter(ua =>
                    ua.includes('iPhone') ||
                    ua.includes('iPad')
                );

            case UserAgentType.BOT:
                return userAgents.filter(ua => ua.includes('bot'));

            case UserAgentType.CHROME:
                return userAgents.filter(ua =>
                    ua.includes('Chrome') &&
                    !ua.includes('Edg') &&
                    !ua.includes('OPR') &&
                    !ua.includes('SamsungBrowser')
                );

            case UserAgentType.FIREFOX:
                return userAgents.filter(ua => ua.includes('Firefox'));

            case UserAgentType.SAFARI:
                return userAgents.filter(ua =>
                    ua.includes('Safari') &&
                    !ua.includes('Chrome')
                );

            case UserAgentType.EDGE:
                return userAgents.filter(ua => ua.includes('Edg'));

            default:
                return userAgents; // 'all' o cualquier otro valor devuelve todos
        }
    }

    /**
     * Devuelve el siguiente user agent en la secuencia y avanza el índice.
     * @returns {string} El siguiente user agent.
     *
     * @example
     * const ua = rotator.getNext();
     * console.log(ua); // 'Mozilla/5.0...'
     */
    getNext() {
        const userAgent = this.userAgents[this.currentIndex];
        this.currentIndex = (this.currentIndex + 1) % this.userAgents.length;
        return userAgent;
    }

    /**
     * Devuelve un user agent aleatorio de la lista actual.
     * @returns {string} User agent aleatorio.
     *
     * @example
     * console.log(rotator.getRandom());
     */
    getRandom() {
        const randomIndex = Math.floor(Math.random() * this.userAgents.length);
        return this.userAgents[randomIndex];
    }

    /**
     * Devuelve el user agent actual sin avanzar el índice.
     * @returns {string} User agent actual.
     *
     * @example
     * console.log(rotator.getCurrent());
     */
    getCurrent() {
        return this.userAgents[this.currentIndex];
    }

    /**
     * Agrega un nuevo user agent a la lista y vuelve a aplicar el filtro actual.
     * @param {string} userAgent - El user agent a agregar.
     *
     * @example
     * rotator.addUserAgent('Mozilla/5.0 (Test)');
     */
    addUserAgent(userAgent) {
        this.allUserAgents.push(userAgent);
        // Refiltrar según el tipo actual
        this.userAgents = this.filterUserAgents(this.allUserAgents, this.type);
    }

    /**
     * Cambia el tipo de filtro actual y vuelve a aplicar el filtro.
     * @param {string} newType - Nuevo tipo de filtro.
     *
     * @example
     * rotator.setType('desktop');
     */
    setType(newType) {
        this.type = newType.toLowerCase();
        this.userAgents = this.filterUserAgents(this.allUserAgents, this.type);
        this.currentIndex = 0; // Reset index
    }

    /**
     * Reinicia el índice al inicio de la lista.
     *
     * @example
     * rotator.reset();
     */
    reset() {
        this.currentIndex = 0;
    }

    /**
     * Devuelve información sobre el estado del rotador.
     * @returns {{type: string, total: number, totalAvailable: number, currentIndex: number}} Objeto con información.
     *
     * @example
     * console.log(rotator.getInfo());
     */
    getInfo() {
        return {
            type: this.type,
            total: this.userAgents.length,
            totalAvailable: this.allUserAgents.length,
            currentIndex: this.currentIndex
        };
    }
}
