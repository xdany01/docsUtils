sequence-fn.js


/**
 * Clase que representa una secuencia numérica.
 * @example
 * const rank = new Sequence();
 *
 * console.log(rank.next()); // 1
 * console.log(rank.next()); // 2
 * console.log(rank.next()); // 3
 */
class Sequence {
    /**
     * Crea una nueva instancia de la clase Sequence.
     * @param {number} increment - El valor por el cual se incrementará la secuencia.
     * @param {number} start - El valor inicial de la secuencia.
     */
    constructor(increment = 1, start = 0) {
        // Inicializa el generador con los parámetros proporcionados.
        this.generator = this.#createGenerator(increment, start);
    }

    /**
     * Generador privado que produce una secuencia de números.
     * @param {number} increment - El valor por el cual se incrementará cada número.
     * @param {number} start - El valor inicial de la secuencia.
     * @yields {number} - El siguiente número en la secuencia.
     */
    * #createGenerator(increment, start) {
        let id = start; // Inicializa el identificador con el valor de inicio.
        while (true) {
            // Produce el siguiente número en la secuencia y lo incrementa.
            yield id += increment;
        }
    }

    /**
     * Devuelve el siguiente valor de la secuencia.
     * @returns {number} - El siguiente número en la secuencia.
     */
    next() {
        // Llama al generador para obtener el siguiente valor.
        return this.generator.next().value;
    }
}
